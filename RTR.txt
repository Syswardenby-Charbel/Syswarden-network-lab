! ============================================
! RTR — Router Configuration
! SysWarden Enterprise Network Lab
! Date: April 27, 2026
! ============================================

hostname RTR

! --- DHCP Excluded Addresses ---
ip dhcp excluded-address 192.168.10.1 192.168.10.10
ip dhcp excluded-address 192.168.20.1 192.168.20.10
ip dhcp excluded-address 192.168.30.1 192.168.30.10
ip dhcp excluded-address 192.168.40.1 192.168.40.10

! --- DHCP Pools ---
ip dhcp pool VLAN10
 network 192.168.10.0 255.255.255.0
 default-router 192.168.10.1

ip dhcp pool VLAN20
 network 192.168.20.0 255.255.255.0
 default-router 192.168.20.1

ip dhcp pool VLAN30
 network 192.168.30.0 255.255.255.0
 default-router 192.168.30.1

ip dhcp pool VLAN40
 network 192.168.40.0 255.255.255.0
 default-router 192.168.40.1

! --- Domain & SSH ---
ip domain-name syswarden.local
ip ssh version 2
username admin privilege 15 secret <PASSWORD>

! --- Subinterfaces (Router-on-a-Stick) ---
interface Ethernet0/0
 no ip address
 ip nat inside

interface Ethernet0/0.10
 encapsulation dot1Q 10
 ip address 192.168.10.1 255.255.255.0
 ip access-group BLOCK-V10-V20 in

interface Ethernet0/0.20
 encapsulation dot1Q 20
 ip address 192.168.20.1 255.255.255.0
 ip access-group BLOCK-V10-V20 in

interface Ethernet0/0.30
 encapsulation dot1Q 30
 ip address 192.168.30.1 255.255.255.0

interface Ethernet0/0.40
 encapsulation dot1Q 40
 ip address 192.168.40.1 255.255.255.0

interface Ethernet0/1
 no ip address
 ip nat outside

! --- NAT ---
ip nat inside source list NAT-VLANS interface Ethernet0/1 overload

ip access-list standard NAT-VLANS
 permit 192.168.10.0 0.0.0.255
 permit 192.168.20.0 0.0.0.255
 permit 192.168.30.0 0.0.0.255
 permit 192.168.40.0 0.0.0.255

! --- ACLs ---
ip access-list extended BLOCK-V10-V20
 deny   ip 192.168.10.0 0.0.0.255 192.168.20.0 0.0.0.255
 deny   ip 192.168.20.0 0.0.0.255 192.168.10.0 0.0.0.255
 permit ip any any

! --- VTY Lines (SSH only) ---
line vty 0 4
 login local
 transport input ssh
